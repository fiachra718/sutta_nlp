SELECT
  s->>'text' AS text,
  COUNT(*)   AS freq
FROM gold_training g
CROSS JOIN LATERAL jsonb_array_elements(g.spans) AS s
WHERE s->>'label' = 'GPE'
GROUP BY s->>'text'
ORDER BY freq DESC;
