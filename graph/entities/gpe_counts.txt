SELECT
  ent->>'text'  AS text,
  ent->>'label' AS label,
  COUNT(*)      AS freq
FROM ati_verses v
CROSS JOIN LATERAL jsonb_array_elements(v.ner_span) AS ent
WHERE ent->>'label' = 'GPE'
GROUP BY ent->>'text', ent->>'label'
ORDER BY freq DESC;
